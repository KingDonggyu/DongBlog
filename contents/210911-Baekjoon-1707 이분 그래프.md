    import sys

    def bfs(start):
        visited[start] = 1
        queue = deque([start])
        while queue:
            v = queue.popleft()

                if visited[i] == 0:
                    visited[i] = -visited[v]

    K = int(input())

        graph = [[] for _ in range(V+1)]
        visited = [0] * (V+1)
        isBipartite = True
        for _ in range(E):
            x, y = map(int, input().split())
            graph[x].append(y)
            graph[y].append(x)

                if not bfs(i):
                    isBipartite = False
                    break
        print("YES" if isBipartite else "NO")
