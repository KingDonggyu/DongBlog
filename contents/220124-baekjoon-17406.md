            copy_A[i+1][c+s-cnt] = copy_A[i][c+s-cnt]

N, M, K = map(int, input().split())
A = [0] + [[0] + [int(i) for i in input().split()] for _ in range(N)]
rot = [[int(i) for i in input().split()] for _ in range(K)]

print(result)
