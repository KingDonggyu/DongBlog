import sys
from itertools import combinations


def calculate(op, v1, v2):
    if op == '+':
        return v1 + v2
    elif op == '-':
        return v1 - v2
    else:
        return v1 * v2


N = int(input())

    if i%2 == 0: c = int(c)
    math.append(c)

bracket_num = [i for i in range(N // 2)]
